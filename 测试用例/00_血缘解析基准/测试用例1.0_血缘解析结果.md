### 原始SQL

```sql
select
    order_no,
    order_date,
    mt.user_id,
    null as repayment_amount,
    if(gmt_success is not null,1,0) as performance_type,
    gmt_success as performance_time,

    ext_flag_map['promotion_from_user_flag'] status,
    ext_flag_map['promotion_single_trade_no'] pay_notify_no,

    ext_flag_map['promotion_deduction_amount'] pay_amount,
    ext_flag_map['promotion_deduction_success_time'] performance_type
from default.mdw_order_v3_international mt
         left join ihotel_default.ods_mkt_peach_promotion_task_merge_da t1
                   on mt.order_no = t1.order_id and t1.dt = '2026-04-21'
where mt.dt = '20260421'
```

## TC-002: 子查询+JOIN+GROUP BY

### 基本信息

| 属性 | 值 |
|------|-----|
| 来源文件 | `测试用例/AB实验/AB_rule底表dwd_abtest_rule_info_di.sql` |
| SQL类型 | INSERT OVERWRITE ... SELECT |
| 复杂度 | ★★☆☆☆ |
| 涉及表数 | 2 |
| CTE数量 | 0 |
| 子查询数量 | 2 |

### 原始SQL

```sql
insert overwrite table ihotel_default.dwd_abtest_rule_info_di 
partition (dt='${zdt.addDay(-1).format("yyyy-MM-dd")}',user_id_type)
select
    ab_exp_id
     ,ab_version
     ,ab_rule_version
     ,device_id as ab_exp_value
     ,user_id_type
from
    (
        select
            ab_exp_id,
            ab_version,
            ab_rule_version,
            case ab_shuntbase
                when 'APP_UID' then 'uid'
                when 'USERID'  then 'user_id'
                end as user_id_type
        from ods_abtest_rule_info
        where dt = '${zdt.addDay(-1).format("yyyyMMdd")}'
          and source = 'hotel'
          and ab_shuntbase in ('APP_UID','USERID')
    ) rule
        join
    (
        select  expid, version, ruleversion, clientcode as device_id, dt,logdate
        from ods_abtest_sdk_log_endtime_hotel
        where dt='${zdt.addDay(-1).format("yyyyMMdd")}'
          and clientcode is not NULL
          and expid is not NULL
          and version is not NULL
          and ruleversion is not NULL
          and expid !=''
          and version !=''
          and clientcode not in ('0','00000000','00000000000000','000000000000000','0000000000000000','0000000000000000000000000000000000000000','','ctrip','elong','352284040670808')
          and (clientcode not like 'tc%' and clientcode not like 'wx%' and clientcode not like 'pd%')
    )ab
    on ab.expid=rule.ab_exp_id and ab.version=rule.ab_version and ab.ruleversion=rule.ab_rule_version
group by 1,2,3,4,5
```

### 血源图结构

```

│  ─────────────────────────────────  │

│        ab_rule_version,             │
│        ab_shuntbase                 │
│  加工: CASE ab_shuntbase →          │
│        user_id_type                 │
└──────────────┬──────────────────────┘
               │ JOIN (ab_exp_id=expid, 
               │      ab_version=version,
               │      ab_rule_version=ruleversion)
               ▼
┌─────────────────────────────────────┐
│   ods_abtest_sdk_log_endtime_hotel  │ (ab子查询)
│  ─────────────────────────────────  │
│  字段: expid, version, ruleversion, │
│        clientcode → device_id       │
└──────────────┬──────────────────────┘
               │
               ▼
┌─────────────────────────────────────┐
│   dwd_abtest_rule_info_di (目标表)   │
│  ─────────────────────────────────  │
│  字段: ab_exp_id, ab_version,       │
│        ab_rule_version,             │
│        ab_exp_value, user_id_type   │
│  分区: dt, user_id_type             │
└─────────────────────────────────────┘
```

### 字段血缘明细

| # | 输出字段 | 加工链路 | 源表 | 源字段 | 加工逻辑 |
|---|----------|----------|------|--------|----------|
| 1 | `ab_exp_id` | ods_abtest_rule_info.ab_exp_id → rule.ab_exp_id → ab_exp_id | ods_abtest_rule_info | ab_exp_id | 直接透传 |
| 2 | `ab_version` | ods_abtest_rule_info.ab_version → rule.ab_version → ab_version | ods_abtest_rule_info | ab_version | 直接透传 |
| 3 | `ab_rule_version` | ods_abtest_rule_info.ab_rule_version → rule.ab_rule_version → ab_rule_version | ods_abtest_rule_info | ab_rule_version | 直接透传 |
| 4 | `ab_exp_value` | ods_abtest_sdk_log_endtime_hotel.clientcode → ab.device_id → ab_exp_value | ods_abtest_sdk_log_endtime_hotel | clientcode | 重命名 |
| 5 | `user_id_type` | ods_abtest_rule_info.ab_shuntbase → CASE判断 → rule.user_id_type → user_id_type | ods_abtest_rule_info | ab_shuntbase | `CASE WHEN 'APP_UID' THEN 'uid' WHEN 'USERID' THEN 'user_id'` |

### 血缘解析要点

- **JOIN条件**: `ab.expid = rule.ab_exp_id AND ab.version = rule.ab_version AND ab.ruleversion = rule.ab_rule_version`（三字段联合JOIN）
- **GROUP BY**: 去重作用，保证输出唯一性
- **分区字段**: `user_id_type` 来自子查询rule的CASE表达式结果

## TC-003: CTE+窗口函数+JOIN

### 基本信息

| 属性 | 值 |
|------|-----|
| 来源文件 | `测试用例/优惠先享需求/dw_order_promotion_deduction_di.sql` |
| SQL类型 | CREATE TABLE + INSERT OVERWRITE |
| 复杂度 | ★★★☆☆ |
| 涉及表数 | 2 |
| CTE数量 | 4 |
| 子查询数量 | 1 |

### 原始SQL

```sql
insert overwrite table ihotel_default.dw_order_promotion_deduction_di
partition (dt='${zdt.addDay(-1).format("yyyy-MM-dd")}')
with task_base as (
    select
        cast(t.order_id as string) as order_no,
        cast(t.performance_note_no as string) as performance_note_no,
        cast(t.order_id as string) as repayment_no,
        cast(t.gmt_success as string) as performance_time,
        row_number() over (
            partition by cast(t.order_id as string)
            order by
                case when t.gmt_success is not null then 0 else 1 end,
                t.gmt_success desc,
                cast(t.performance_note_no as string) desc
        ) as rn
    from ihotel_default.ods_mkt_peach_promotion_task_merge_da t
    where t.dt = '${zdt.addDay(-1).format("yyyyMMdd")}'
      and t.order_id is not null
),
task_dedup as (
    select order_no, performance_note_no, repayment_no, performance_time
    from task_base
    where rn = 1
),
order_base as (
    select order_no, order_date, user_id, repayment_type, status,
           pay_notify_no, pay_time, pay_source, pay_source1, pay_amount
    from (
        select
            cast(o.order_no as string) as order_no,
            cast(o.order_date as string) as order_date,
            cast(o.user_id as string) as user_id,
            cast(coalesce(o.ext_flag_map['active_promotion_flag'],
                          o.ext_flag_map['activePromotionFlag']) as string) as repayment_type,
            cast(coalesce(o.ext_flag_map['promotion_from_user_flag'],
                          o.ext_flag_map['promotionFromUserFlag']) as string) as status,
            cast(coalesce(o.ext_flag_map['promotion_single_trade_no'],
                          o.ext_flag_map['promotionSingleTradeNo']) as string) as pay_notify_no,
            cast(coalesce(o.ext_flag_map['promotion_deduction_success_time'],
                          o.ext_flag_map['promotionDeductionSuccessTime']) as string) as pay_time,
            cast(coalesce(o.ext_flag_map['promotion_deduction_pay_brand_name'],
                          o.ext_flag_map['promotionDeductionPayBrandName']) as string) as pay_source,
            cast(coalesce(o.ext_flag_map['promotion_deduction_pay_method_name'],
                          o.ext_flag_map['promotionDeductionPayMethodName']) as string) as pay_source1,
            cast(coalesce(o.ext_flag_map['promotion_deduction_amount'],
                          o.ext_flag_map['promotionDeductionAmount']) as string) as pay_amount,
            row_number() over (
                partition by cast(o.order_no as string)
                order by cast(o.order_date as string) desc
            ) as rn
        from default.mdw_order_v3_international o
        join (select distinct order_no from task_dedup) k
            on cast(o.order_no as string) = k.order_no
    ) s
    where rn = 1
)
select
    t.order_no, o.order_date, o.user_id,
    t.performance_note_no, t.repayment_no,
    cast(null as string) as repayment_amount,
    case when t.performance_time is not null then '0' else '1' end as performance_type,
    t.performance_time, o.repayment_type,
    cast(null as string) as over_due,
    cast(null as string) as bad_debt,
    o.status, o.pay_notify_no, o.pay_time,
    o.pay_source, o.pay_source1, o.pay_amount
from task_dedup t
left join order_base o on t.order_no = o.order_no;
```

| 10 | `status` | mdw_order_v3_international.ext_flag_map['promotion_from_user_flag'] 或 ['promotionFromUserFlag'] → COALESCE → CAST → order_base → **输出** | mdw_order_v3_international | ext_flag_map | `COALESCE(两个key)` + CAST |
| 11 | `pay_notify_no` | mdw_order_v3_international.ext_flag_map['promotion_single_trade_no'] 或 ['promotionSingleTradeNo'] → COALESCE → CAST → order_base → **输出** | mdw_order_v3_international | ext_flag_map | `COALESCE(两个key)` + CAST |

### 血缘解析要点

- **CTE链路**: `ods_mkt_* → task_base(RN) → task_dedup(WHERE rn=1) → 最终JOIN`
- **窗口函数**: `ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY ...)` 用于去重
- **COALESCE**: 处理Map字段命名风格兼容（驼峰 vs 下划线）
- **JOIN关系**: task_dedup LEFT JOIN order_base，以任务表为主表

## TC-004: 双CTE+复杂聚合

### 基本信息

| 属性 | 值 |
|------|-----|
| 来源文件 | `测试用例/AB实验/搜索数据分流-device_id-dw_ihotel_abtest_detail_searchlist_di.sql` |
| SQL类型 | SELECT |
| 复杂度 | ★★★☆☆ |
| 涉及表数 | 1（自引用） |
| CTE数量 | 2 |
| 子查询数量 | 0 |

### 原始SQL（核心部分）

```sql
with search_result as (
    select
        a.dt,
        count(distinct a.search_request_uid) as search_times,
        count(distinct a.orig_device_id) as search_uv,
        count(case when a.is_display = '1' then a.hotel_seq else null end) as show_item,
        -- ... 更多聚合指标
    from ihotel_default.dw_ihotel_aa_index_searchlist_di a
    where dt = '2026-04-12' and type = 'searchlist'
    group by 1
),
order_result as (
    select
        a.dt,
        sum(a.order_init_gmv) as total_gmv,
        sum(a.order_room_night) as total_room_night,
        sum(a.order_init_gmv)/sum(a.order_room_night) as order_adr,
        sum(a.order_order_commission) as total_order_commission
    from ihotel_default.dw_ihotel_aa_index_searchlist_di a
    where dt = '2026-04-12' and type = 'order'
    group by 1
)
select
    a.dt,
    cast(b.total_order_commission / a.show_uv as decimal(20,2)) as revenue_per_uv,
    concat(round(a.click_uv / a.show_uv * 100, 2), '%') as s2d,
    -- ... 更多计算指标
from search_result a
join order_result b on a.dt = b.dt
```

### 原始SQL（核心部分）

```sql
with search_result as (

        a.datee, a.ab_exp_id, a.ab_version, a.ab_rule_version,

          and newolduser = '新客' and highlowstar = 'low_star'
          and is_big_order_user = '正常用户'
          and country_name = '韩国' and city_name = '首尔'
    ) a
    group by 1,2,3,4
),
order_result as (
    select 
        a.order_datee datee, a.ab_exp_id, a.ab_version, a.ab_rule_version,
        sum(a.order_init_gmv) as total_gmv,
        sum(a.order_room_night) as total_room_night,
        sum(a.order_init_gmv)/sum(a.order_room_night) as order_adr,
        sum(a.order_order_commission) as total_order_commission
    from (
        select * from ihotel_default.dw_ihotel_abtest_index_searchlist_di
        where dt = '2026-04-14' and type = 'order'
          and user_id_type = 'uid'
          and ab_exp_id = '260402_ho_gj_comment_limit'
          and newolduser = '新客' and highlowstar = 'low_star'
          and is_big_order_user = '正常用户'
          and country_name = '韩国' and city_name = '首尔'
    ) a
    group by 1,2,3,4
)
select
    a.datee, a.ab_version, a.ab_rule_version,
    cast(b.total_order_commission / a.show_uv as decimal(20,2)) as revenue_per_uv,
    concat(round(a.click_uv / a.show_uv * 100, 2), '%') as s2d,
    -- ... 更多计算指标
from search_result a
join order_result b
    on a.datee = b.datee
        and a.ab_exp_id = b.ab_exp_id
        and a.ab_version = b.ab_version
        and a.ab_rule_version = b.ab_rule_version
```
