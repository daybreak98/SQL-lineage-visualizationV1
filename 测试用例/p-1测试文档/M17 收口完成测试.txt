1.解析以下sql时报错缺少元数据：
UNKNOWN_TABLE
Reason: 元数据中找不到表: default.default.t3
Impact: {"catalog":"default","schema":"default","table":"t3","alias":"t3"}
Action: 请先导入对应表元数据，或检查 SQL 中的 catalog/schema/table

WITH mt AS (
  SELECT
    agent_id,
    DATE_FORMAT(CAST(order_time AS TIMESTAMP), 'yyyyMM') AS dd,
    SUM(
      order_amount * CASE WHEN currency = '美元' THEN 5 WHEN currency = '欧元' THEN 7 ELSE 1 END
    ) AS gmv
  FROM intl_hotel_orders
  GROUP BY
    1,
    2
), t1 AS (
  SELECT
    agent_id,
    dd,
    gmv,
    IF(dd = MIN(dd) OVER (PARTITION BY agent_id ORDER BY dd), gmv, 0) AS GMV_min,
    IF(dd = MAX(dd) OVER (PARTITION BY agent_id ORDER BY dd DESC), gmv, 0) AS GMV_max
  FROM mt
), t2 AS (
  SELECT
    agent_id,
    gmv,
    SUM(GMV_max) / SUM(GMV_min) AS gmv_rate
  FROM t1
  GROUP BY
    1,
    2
), t3 AS (
  SELECT
    agent_id,
    gmv,
    ROW_NUMBER() OVER (PARTITION BY agent_id ORDER BY gmv_rate DESC) AS rnk
  FROM t2
)
SELECT
  agent_id,
  gmv
FROM t3
WHERE
  rnk <= 3
